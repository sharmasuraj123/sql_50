SELECT
    ROUND(
        COUNT(distinct a.player_id)/(select COUNT(distinct player_id) from Activity),
        2
    ) AS fraction
FROM Activity a
JOIN (
    SELECT player_id, MIN(event_date) AS first_login
    FROM Activity
    GROUP BY player_id
) b
ON a.player_id = b.player_id
WHERE DATEDIFF(a.event_date, b.first_login) = 1;
